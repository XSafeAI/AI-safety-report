# -----------------------------------------------
# Gemini API (Google AI)
# -----------------------------------------------
# 获取地址: https://aistudio.google.com/app/apikey
# 支持模型: gemini-3-pro-preview, gemini-2.5-pro, gemini-2.5-flash

# -----------------------------------------------
# OpenAI API
# -----------------------------------------------
# 获取地址: https://platform.openai.com/api-keys
# 支持模型: gpt-5, gpt-5-mini, gpt-4.1, gpt-4o, gpt-4o-mini
OPENAI_API_KEY=your-openai-api-key-here

# -----------------------------------------------
# DeepSeek API
# -----------------------------------------------
# 获取地址: https://platform.deepseek.com/api_keys
# 支持模型: deepseek-reasoner, deepseek-chat

# -----------------------------------------------
# DashScope API (阿里云通义千问)
# -----------------------------------------------
# 获取地址: https://dashscope.console.aliyun.com/apiKey

# -----------------------------------------------
# Ark API (火山引擎豆包)
# -----------------------------------------------
# 获取地址: https://console.volcengine.com/ark/region:ark+cn-beijing/apiKey
# 支持模型: doubao-seed-1-6-251015, doubao-seed-1-6-vision-250815
ARK_API_KEY=your-ark-api-key-here

# -----------------------------------------------
# XAI API (Grok by xAI)
# -----------------------------------------------
# 获取地址: https://console.x.ai/
# 支持模型: grok-4-1-fast-non-reasoning, grok-4-1-reasoning

# -----------------------------------------------
# SiliconFlow API
# -----------------------------------------------
# 获取地址: https://cloud.siliconflow.cn/account/ak
# 支持模型: 多种开源模型
